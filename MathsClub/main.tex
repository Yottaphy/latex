                s &= r\theta \\
                A&=\frac{1}{2}r\times\left(r\theta\right) \\          
                A&=\frac12 \theta r^2 \\
            \end{align*}
            \begin{align*}
                \frac{\Delta A}{\Delta t} &= \frac{1}{2}\frac{\Delta \theta}{\Delta t}r^2 \\
                \frac{\Delta A}{\Delta t} &= \frac{1}{2}r^2 \omega =\frac{L}{2m}  \\          
                \frac{\Delta A}{\Delta t} &= \rm{constant}
            \end{align*}
